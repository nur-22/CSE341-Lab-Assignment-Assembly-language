msg1 DB "Enter three initials: $"

.CODE 
MAIN PROC 
;iniitialize DS

;Code here

;DISPLAYING STRING
LEA DX, msg1

INT 21H

;INPUT
MOV AH, 1 ;STORED IN AL first val
INT 21H 

MOV BL, AL

MOV AH, 1 ;STORED IN AL middle val
INT 21H 

MOV CL, AL

INT 21H

;CURSOR IN THE NEXT LINE
MOV AH, 2

;OUTPUT
MOV DL, BL ;HERE CL=AL

INT 21H 

;OUTPUT
MOV DL, TRL ;HERE CL=AL
